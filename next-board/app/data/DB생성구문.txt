board 테이블생성

1. root접속속
mysql -u root -p

2. DB 생성
CREATE DATABASE react_board;

3.user 생성
CREATE USER IF NOT EXISTS 'reactUser'@'localhost' IDENTIFIED BY 'mysql';
GRANT ALL PRIVILEGES ON react_board.* TO 'reactUser'@'localhost';
FLUSH PRIVILEGES;

mysql -u reactUser -p
USE react_board;

CREATE TABLE board (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    contents TEXT,
    writer VARCHAR(50) NOT NULL,
    reg_date DATETIME DEFAULT CURRENT_TIMESTAMP
);
